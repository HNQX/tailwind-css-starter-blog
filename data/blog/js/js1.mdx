**01、[去抖函数](https://www.zhihu.com/search?q=%E5%8E%BB%E6%8A%96%E5%87%BD%E6%95%B0&search_source=Entity&hybrid_search_source=Entity&hybrid_search_extra=%7B%22sourceType%22%3A%22answer%22%2C%22sourceId%22%3A3280879891%7D)以限制它被调用的次数.**

```js
function debounce(func, delay) {
  let timer;
  return function() {
    clearTimeout(timer);
    timer = setTimeout(func, delay);
  };
}

```

**02、[节流函数](https://www.zhihu.com/search?q=%E8%8A%82%E6%B5%81%E5%87%BD%E6%95%B0&search_source=Entity&hybrid_search_source=Entity&hybrid_search_extra=%7B%22sourceType%22%3A%22answer%22%2C%22sourceId%22%3A3280879891%7D)以限制调用它的速率**：

```js
function throttle(func, limit) {
  let throttled = false;
  return function() {
    if (!throttled) {
      func();
      throttled = true;
      setTimeout(function() {
        throttled = false;
      }, limit);
    }
  };
}

```

**09、检查一个数是否为[质数](https://www.zhihu.com/search?q=%E8%B4%A8%E6%95%B0&search_source=Entity&hybrid_search_source=Entity&hybrid_search_extra=%7B%22sourceType%22%3A%22answer%22%2C%22sourceId%22%3A3280879891%7D)：**

```js
function isPrime(number) {
  if (number <= 1) {
    return false;
  }
  for (let i = 2; i <= Math.sqrt(number); i++) {
    if (number % i === 0) {
      return false;
    }
  }
  return true;
}

```

  let result = '';
  const characters = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789';
  for (let i = 0; i < length; i++) {
